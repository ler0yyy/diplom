Вставьте этот код в объект `ThisPresentation` внутри PowerPoint VBA Editor (а не в .bas модуль).

Он будет вызывать макросы при смене слайда в режиме показа.

---

Option Explicit

Private WithEvents ppApp As Application

Private Sub Presentation_Open()
    Set ppApp = Application
End Sub

Private Sub ppApp_SlideShowNextSlide(ByVal Wn As SlideShowWindow)
    On Error GoTo SafeExit

    Dim sld As Slide
    Set sld = Wn.View.Slide

    ' Вариант 1: управлять по Notes:
    ' В заметках слайда (Notes) напишите:
    '   [POLL slide_id=123]
    ' или
    '   [STATS]
    '
    Dim notesText As String
    notesText = ""
    On Error Resume Next
    notesText = sld.NotesPage.Shapes.Placeholders(2).TextFrame.TextRange.Text
    On Error GoTo 0

    If InStr(1, notesText, "[POLL", vbTextCompare) > 0 Then
        Dim slideId As Long
        slideId = ExtractLong(notesText, "slide_id=")
        If slideId > 0 Then
            CreateSessionForSlide slideId
        End If
    ElseIf InStr(1, notesText, "[STATS", vbTextCompare) > 0 Then
        RefreshStatsFromSession
    End If

SafeExit:
End Sub

Private Function ExtractLong(ByVal text As String, ByVal key As String) As Long
    Dim p As Long
    p = InStr(1, text, key, vbTextCompare)
    If p = 0 Then Exit Function

    Dim s As String
    s = Mid$(text, p + Len(key))

    Dim i As Long
    Dim digits As String
    digits = ""
    For i = 1 To Len(s)
        Dim ch As String
        ch = Mid$(s, i, 1)
        If ch >= "0" And ch <= "9" Then
            digits = digits & ch
        Else
            Exit For
        End If
    Next i

    If digits <> "" Then ExtractLong = CLng(digits)
End Function
